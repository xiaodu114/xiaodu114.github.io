冒泡排序

function bubbleSort(arr) {
    let len = arr.length;
    let isSwitch;
    for (let i = 0; i < len - 1; i++) {
        isSwitch = false;
        for (let j = 0; j < len - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                const temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                isSwitch = true;
            }
        }
        if (!isSwitch) break;
    }
    return arr;
}
