栗子 1

function makePower(power) {
    return function (num) {
        //return Math.pow(num, power);
        return num ** power;
    };
}

let power2Fn = makePower(2);
let power3Fn = makePower(3);

console.log(power2Fn(4));
console.log(power2Fn(5));
console.log(power3Fn(2));
console.log(power3Fn(3));
