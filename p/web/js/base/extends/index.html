临时记录

//  父类：Person
function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.say = function () {
    console.log(`父类方法：大家好，我叫${this.name}，今年${this.age}了……`);
};
var p1 = new Person("xiaodu114", 18);
p1.say();


//  子类：Student
function Student(name, age, school) {
    Person.call(this, name, age);
    this.school = school;
}
//  设置子类的 prototype
// //  方式一
// Student.prototype = {
//     constructor: Student,
//     __proto__: Person.prototype
// };
// //  方式二
// Student.prototype = Object.create(Person.prototype);
// Student.prototype.constructor = Student;
//  方式三
Object.setPrototypeOf(Student.prototype, Person.prototype);
//  重写父类方法;
Student.prototype.say = function () {
    console.log(`子类方法：大家好，我是来自${this.school}的${this.name}，今年${this.age}了……`);
};
console.log("Student.__proto__ 和 Function.prototype 是否相等：", Student.__proto__ === Function.prototype);
console.log("Student.prototype.constructor 和 Student 是否相等：", Student.prototype.constructor === Student);
var stu1 = new Student("小杜001", 20, "哈工大");
stu1.say();
console.log(stu1.__proto__, Student.prototype, stu1.__proto__ === Student.prototype);
